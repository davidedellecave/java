<plugin>
				<groupId>com.google.code.maven-replacer-plugin</groupId>
				<artifactId>replacer</artifactId>
				<version>1.5.3</version>
				<executions>
					<execution>
						<phase>prepare-package</phase>
						<goals>
							<goal>replace</goal>
						</goals>
					</execution>
				</executions>
				<configuration>
					<file>src/main/resources/version.conf</file>
					<outputFile>src/main/webapp/version.txt</outputFile>
					<regex>false</regex>
					<replacements>
						<replacement>
							<token>%buildtime%</token>
							<value>${maven.build.custom.timestamp}</value>
						</replacement>
						<replacement>
							<token>%version%</token>
							<value>${project.version}</value>
						</replacement>
					</replacements>
				</configuration>
			</plugin>
